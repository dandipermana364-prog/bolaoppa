    function escapeHTML(value) {
      return String(value ?? "")
        .replace(/&/g, "&amp;")
        .replace(/</g, "&lt;")
        .replace(/>/g, "&gt;")
        .replace(/"/g, "&quot;")
        .replace(/'/g, "&#039;");
    }

    function parseMatchLine(line) {
      const clean = line.replace(/\s+/g, " ").trim();
      const regex = /^(\d{1,2}\/\d{1,2}\s+\d{1,2}:\d{2}\s*(?:WIB|WITA|WIT)?)\s+(.+?)\s+vs\s+(.+?)(?:\s+((?:\d+(?:[\/.]\d+)?|\d+)\s*:\s*(?:\d+(?:[\/.]\d+)?|\d+)))?$/i;
      const found = clean.match(regex);

      if (found) {
        return {
          time: found[1].trim(),
          home: found[2].trim(),
          away: found[3].trim(),
          score: found[4] ? found[4].replace(/\s*:\s*/, " : ").trim() : "VS"
        };
      }

      const parts = clean.split(/\s+vs\s+/i);
      if (parts.length >= 2) {
        const left = parts[0].trim();
        const right = parts.slice(1).join(" vs ").trim();
        const timeMatch = left.match(/^(\d{1,2}\/\d{1,2}\s+\d{1,2}:\d{2}\s*(?:WIB|WITA|WIT)?)/i);
        const scoreMatch = right.match(/((?:\d+(?:[\/.]\d+)?|\d+)\s*:\s*(?:\d+(?:[\/.]\d+)?|\d+))$/);
        const time = timeMatch ? timeMatch[1].trim() : "-";
        const home = timeMatch ? left.replace(timeMatch[1], "").trim() : left;
        const score = scoreMatch ? scoreMatch[1].replace(/\s*:\s*/, " : ").trim() : "VS";
        const away = scoreMatch ? right.replace(scoreMatch[1], "").trim() : right;

        return { time, home, away, score };
      }

      return { time: "-", home: clean, away: "-", score: "VS" };
    }

    function parseInputData(raw) {
      const lines = raw
        .split(/\r?\n/)
        .map(line => line.trim())
        .filter(Boolean);

      const leagues = [];
      let currentLeague = null;

      lines.forEach(line => {
        if (/\s+vs\s+/i.test(line)) {
          if (!currentLeague) {
            currentLeague = { name: "PREDIKSI BOLA", matches: [] };
            leagues.push(currentLeague);
          }
          currentLeague.matches.push(parseMatchLine(line));
        } else {
          currentLeague = { name: line, matches: [] };
          leagues.push(currentLeague);
        }
      });

      return leagues.filter(league => league.matches.length > 0);
    }

    function generateDashboard() {
      const dateVal = document.getElementById("inputDate").value;
      const marqueeVal = document.getElementById("inputMarquee").value;
      const dataVal = document.getElementById("inputData").value;

      if (!dataVal.trim()) {
        alert("Masukkan data pertandingan terlebih dahulu!");
        return;
      }

      const leagues = parseInputData(dataVal);
      if (!leagues.length) {
        alert("Data belum terbaca. Pastikan format match memakai kata 'vs', contoh: 30/05 00:00 WIB Team A vs Team B 1 : 2");
        return;
      }

      const html = buildDashboardHTML(dateVal, marqueeVal, leagues);
      document.getElementById("outputHTML").value = html;
      document.getElementById("livePreview").innerHTML = html;
    }

    // Fungsi ini dipakai agar dropdown filter tetap jalan di live preview generator.
    // Kode hasil generate juga sudah punya fungsi filter sendiri saat ditempel di website.
    function tgFilter(value) {
      document.querySelectorAll('#livePreview #tgup-dash-pro .tg-group').forEach(group => {
        group.style.display = value === 'all' || group.id === value ? 'block' : 'none';
      });
    }

    async function copyCode() {
      const output = document.getElementById("outputHTML");
      if (!output.value) {
        alert("Generate kode terlebih dahulu.");
        return;
      }

      try {
        await navigator.clipboard.writeText(output.value);
        alert("Kode HTML berhasil disalin!");
      } catch (err) {
        output.select();
        document.execCommand("copy");
        alert("Kode HTML berhasil disalin!");
      }
    }
